from tkinter import *
# Window Create
window = Tk()
window.minsize(height=600, width=600)
window.config(padx=20, pady=20)


# Label
label = Label(text="My Label", font=('Arial', 10, 'bold'))
# label.pack()
label.grid(column=0, row=0)


# Entry
user_input = Entry(width=10)
user_input.insert(END, string="RISHABH")
# user_input.pack()
user_input.grid(column=0, row=1)
# value = user_input.get()


# Button
def button_clicked():
    print("Clicked")
    # label.config(text="Button got clicked")
    label['text'] = user_input.get()


button = Button(text="Click", command=button_clicked)
# button.pack()
button.grid(column=0, row=2)


button1 = Button(text="Click", command=button_clicked)
# button.pack()
button1.grid(column=2, row=0)


# Text area
text = Text(height=5, width=10)
text.insert(END, "Multi Line text area")
# text.pack()
text.grid(column=0, row=3)


# Spinbox
def spinbox():
    print(spin.get())


spin = Spinbox(from_=0, to=20, width=5, command=spinbox)
# spin.pack()
spin.grid(column=0, row=4)


# Scale
def scale(value):
    print(value)


scale = Scale(from_=0, to=100, command=scale)
# scale.pack()
scale.grid(column=0, row=5)


# Check Box
def check_button():
    print(checked_state.get())


checked_state = IntVar()
checkbutton = Checkbutton(text="ON", variable=checked_state, command=check_button)
# checkbutton.pack()
checkbutton.grid(column=0, row=6)


# Radio Button
def radio_click():
    print(radio_checked.get())


radio_checked = IntVar()
radiobutton1 = Radiobutton(text='Option 1', value=1, variable=radio_checked, command=radio_click)
radiobutton2 = Radiobutton(text='Option 2', value=2, variable=radio_checked, command=radio_click)
# radiobutton1.pack()
# radiobutton2.pack()
radiobutton1.grid(column=0, row=7)
radiobutton2.grid(column=0, row=8)


# List Box
def listbox_used(event):
    print(event)
    print(listbox.get(listbox.curselection()))


listbox = Listbox(height=4)
fruits = ['Apple', 'Mango', 'Orange', 'Avocado']
for item in fruits:
    listbox.insert(fruits.index(item), item)
listbox.bind("<<ListboxSelect>>", listbox_used)
# listbox.pack()
listbox.grid(column=0, row=9)


window.mainloop()
